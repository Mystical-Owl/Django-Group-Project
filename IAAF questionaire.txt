calculate_iaaf <- function(age, income, dependents, marital, cost_ratio) {
  income_ratio <- min(income/50000, 1)  # HK median benchmark
  dep_adjust <- 1 - (dependents * 0.15)
  age_factor <- ifelse(age < 50, 0.9, 0.7)
  cost_burden <- 1 - cost_ratio
  
  score <- 25*income_ratio + 30*dep_adjust + 20*age_factor + 25*cost_burden
  tier <- ifelse(score > 70, "Proceed", ifelse(score > 40, "Caution", "Decline"))
  return(list(score = round(score), tier = tier, max_premium_pct = max(0.03, score/2000)))
}

# Subject example
